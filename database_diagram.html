🔗 Foreign Key Relationships

products → categories

Foreign Key: products.category_id → categories.id

Type: Many-to-One (Many products belong to one category)

Action: ON DELETE SET NULL

products → suppliers

Foreign Key: products.supplier_id → suppliers.id

Type: Many-to-One (Many products from one supplier)

Action: ON DELETE SET NULL

sales → users

Foreign Key: sales.user_id → users.id

Type: Many-to-One (Cashier processes many sales)

Action: ON DELETE SET NULL

sales → customers

Foreign Key: sales.customer_id → customers.id

Type: Many-to-One (Customer makes many sales)

Action: ON DELETE SET NULL

sale_items → sales

Foreign Key: sale_items.sale_id → sales.id

Type: Many-to-One (One sale has many items)

Action: ON DELETE CASCADE

sale_items → products

Foreign Key: sale_items.product_id → products.id

Type: Many-to-One (Product appears in many sales)

Action: ON DELETE SET NULL

purchase_orders → suppliers

Foreign Key: purchase_orders.supplier_id → suppliers.id

Type: Many-to-One (Supplier has many orders)

Action: ON DELETE SET NULL

purchase_orders → users

Foreign Key: purchase_orders.user_id → users.id

Type: Many-to-One (User creates many orders)

Action: ON DELETE SET NULL

stock_movements → products

Foreign Key: stock_movements.product_id → products.id

Type: Many-to-One (Product has many stock movements)

Action: ON DELETE CASCADE

stock_movements → users

Foreign Key: stock_movements.user_id → users.id

Type: Many-to-One (User records many stock movements)

Action: ON DELETE SET NULL

📖 Legend

🔑 PK = Primary Key
🔗 FK = Foreign Key
🏷️ UK = Unique Key
📌 ON DELETE CASCADE = Delete related records automatically
📌 ON DELETE SET NULL = Set foreign key to NULL when parent deleted
